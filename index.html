// LIVE DASHBOARD KUTULARI (KACHEL) ÜRETİMİ function buildDashboardView() { const container = document.getElementById('reportsContainer'); if(!container) return; container.innerHTML = ""; // Sadece seçili olan Schicht verilerini filtrele const filtered = cachedDashboardData.filter(r => r.schicht === activeFilterSchicht); if(filtered.length === 0) { container.innerHTML = `
Keine Meldungen für Schicht ${activeFilterSchicht} vorhanden.
`; return; } filtered.forEach(rep => { const box = document.createElement('div'); box.className = 'report-box'; // Dinamik gelen Ausschuss ve Störung satırlarını kutuya basmak için hazırla let ausHtml = ''; if(rep.ausDetails && rep.ausDetails.length > 0) { ausHtml = `
Ausschuss Details:
    ` + rep.ausDetails.map(d => `
  • • ${d.code} -> ${d.qty} Stk
    • `).join('') + `
`; } let stoerHtml = ''; if(rep.stoerDetails && rep.stoerDetails.length > 0) { stoerHtml = `
Störungen:
    ` + rep.stoerDetails.map(s => `
  • • ${s.code} -> ${s.qty} min
    • `).join('') + `
`; } box.innerHTML = `
${rep.anlage}
${rep.datum} - ${rep.schichtType}
Art-Nr: ${rep.artikelnummer || 'N/A'}
  • Gut: ${rep.gut || 0} Stk
  • Ausschuss Gesamt: ${rep.ausschussGesamt || 0} Stk
${ausHtml} ${stoerHtml}
👤 Erfasst von: ${rep.user || 'System'}
`; container.appendChild(box); }); }
